14
15 size_t w_min = *(prefixSumLoadEnd - 1) / num_segments;
16 size_t w_max = 2*w_min + 1;

20 /* probe function */
21 auto probe_load_sharing = [&](size_t wprobe) {
22 /* separators */
23 size_t last_separator = 0;
24 size_t last_separator_value = 0;

26 /* find separators */
27 for (size_t separator = 0; separator < num_segments - 1; ++separator) {
28 if (*(prefixSumLoadEnd - 1) - last_separator_value > wprobe * (num_segments - separator))
29 return false;
30

33 std::upper_bound(prefixSumLoadBegin + last_separator, prefixSumLoadEnd, wprobe + last_separator_value)) - 1;
34
35 /* prepare next iteration */
36 last_separator_value = prefixSumLoadBegin[last_separator];
37 separators[separator] = last_separator;
38 }
39
40 /* check last segment */
41 if (*(prefixSumLoadEnd - 1) - last_separator_value > wprobe)
42 return false;
43
44 /* copy separators over */
45 for (size_t i = 0; i < num_segments - 1; ++i)
46 workSharingIndexesBegin[i + 1] = separators[i] - 1;
47
48 return true;
49 };
50
51 /* check if their is a single segment */
52 if (num_segments > 1) {
53
54 /* dichotomic search */
55 // "inverse" dichotomic search to find the upper bound
56 while(!probe_load_sharing(w_max)) { w_max *= 2; };
57
58 // actual dichotomic search
59 while (w_max - w_min > 1) {
60 size_t middle = (w_min + w_max) / 2;
61
62 if (probe_load_sharing(middle)) {
63 w_max = middle;

66 }
