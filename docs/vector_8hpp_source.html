98 /*
99 "upsize_policy" is a multiplier (>1) that forces any upsize to add a margin to avoid frequent resize.
100 "downsize_policy" is a multiplier (<1) that forces a down_size to free memory only if the freed memory exceed the downsize_policy
101 (to allow memory to be freed and given back to another vector).
102 "min_state_size" is the minimum size of a vector, to avoid small vectors which are bound to be resized frequently.
103 */
105 void resize(size_t n, const uint align_byte_length_=std::alignment_of<T>()) const {
106 n = std::max(min_vector_size, n); // never resize under min_vector_size
107
108 if (size_ < n || // resize if we absolutely have to because the state won't fit
109 n*upsize_policy < size_*downsize_policy) { // resize if the size we resize to is small enough (to free memory)
110 // for later allignment
111 size_t old_size_ = size_;
112
113 size_ = n*upsize_policy;
114 int offset = std::distance(unaligned_ptr, ptr);
115 unaligned_ptr = (T*)realloc(unaligned_ptr, (size_ + align_byte_length_)*sizeof(T));
116
117 if (unaligned_ptr == NULL)
118 throw std::runtime_error("bad allocation in fast_vector !!");
119
120 ptr = unaligned_ptr + offset;
121 if (align_byte_length_ > 1) {
122 // manual allignment:
123 size_t allign_offset = ((size_t)ptr)%align_byte_length_;
124
125 if (allign_offset != 0)
126 // allign by rotating to the left:
127 if (allign_offset <= offset) {
128 std::rotate<char*>(((char*)ptr) - allign_offset, (char*)ptr, ((char*)ptr) + old_size_*sizeof(T));
129 ptr -= allign_offset;
130 } else {
131 // allign by rotating to the right
132 allign_offset = align_byte_length_ - allign_offset;
133 std::rotate<char*>((char*)ptr, ((char*)ptr) + old_size_*sizeof(T), ((char*)ptr) + old_size_*sizeof(T) + allign_offset);
134 ptr += allign_offset;
135 }
136 }
137 }
138 }
