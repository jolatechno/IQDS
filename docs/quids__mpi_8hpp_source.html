108 void send_objects(size_t num_object_sent, int node, MPI_Comm communicator, bool send_num_child=false) {
109 const static size_t max_int = 1 << 31 - 1;
110
111 /* send size */
112 MPI_Send(&num_object_sent, 1, MPI_UNSIGNED_LONG_LONG, node, 0 /* tag */, communicator);
113 if (num_object_sent == 0)
114 return;
115 size_t begin = num_object - num_object_sent;
116 size_t send_object_size = object_begin[num_object] - object_begin[begin];
117 MPI_Send(&send_object_size, 1, MPI_UNSIGNED_LONG_LONG, node, 0 /* tag */, communicator);
118
119 /* verify send */
120 bool send;
121 MPI_Recv(&send, 1, MPI_UNSIGNED_LONG_LONG, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
122
123 if (send) {
124 /* prepare send */
125 size_t send_object_begin = object_begin[begin];
126 #pragma omp parallel for
127 for (size_t i = begin + 1; i <= num_object; ++i)
128 object_begin[i] -= send_object_begin;
129
130 /* send properties */
131 MPI_Send(&magnitude[begin], num_object_sent, mag_MPI_Datatype, node, 0 /* tag */, communicator);
132 MPI_Send(&object_begin[begin + 1], num_object_sent, MPI_UNSIGNED_LONG_LONG, node, 0 /* tag */, communicator);
133 MPI_Send(&object_size[begin], num_object_sent, MPI_UNSIGNED, node, 0 /* tag */, communicator);
134
135 /* send objects */
136 size_t send_object_size = object_begin[num_object];
137 while (send_object_size > max_int) {
138 MPI_Send(&objects[send_object_begin], max_int, MPI_CHAR, node, 0 /* tag */, communicator);
139
140 send_object_size -= max_int;
141 send_object_begin += max_int;
142 }
143
144 MPI_Send(&objects[send_object_begin], send_object_size, MPI_CHAR, node, 0 /* tag */, communicator);
145
146 if (send_num_child)
147 /* send num child */
148 MPI_Send(&num_childs[begin], num_object_sent, MPI_UNSIGNED, node, 0 /* tag */, communicator);
149
150 /* pop */
151 pop(num_object_sent, false);
152 }
153 }

161 void receive_objects(int node, MPI_Comm communicator, bool receive_num_child=false, size_t max_mem=-1) {
162 const static size_t max_int = 1 << 31 - 1;
163
164 /* receive size */
165 size_t num_object_sent, send_object_size;
166 MPI_Recv(&num_object_sent, 1, MPI_UNSIGNED_LONG_LONG, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
167 if (num_object_sent == 0)
168 return;
169 MPI_Recv(&send_object_size, 1, MPI_UNSIGNED_LONG_LONG, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
170
171 /* verify memory limit */
172 static const size_t iteration_memory_size = ITERATION_MEMORY_SIZE;
173 bool recv = num_object_sent*iteration_memory_size + send_object_size < max_mem;
174 MPI_Send(&recv, 1, MPI_CHAR, node, 0 /* tag*/, communicator);
175
176 if (recv) {
177 /* prepare state */
178 size_t send_object_begin = object_begin[num_object];
179 resize(num_object + num_object_sent);
180 allocate(send_object_begin + send_object_size);
181
182 /* receive properties */
183 MPI_Recv(&magnitude[num_object], num_object_sent, mag_MPI_Datatype, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
184 MPI_Recv(&object_begin[num_object + 1], num_object_sent, MPI_UNSIGNED_LONG_LONG, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
185 MPI_Recv(&object_size[num_object], num_object_sent, MPI_UNSIGNED, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
186
187 /* receive objects */
188 size_t object_offset = send_object_begin;
189 while (send_object_size > max_int) {
190 MPI_Recv(&objects[send_object_begin], max_int, MPI_CHAR, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
191
192 send_object_size -= max_int;
193 send_object_begin += max_int;
194 }
195
196 MPI_Recv(&objects[send_object_begin], send_object_size, MPI_CHAR, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
197
198 /* correct values */
199 #pragma omp parallel for
200 for (size_t i = num_object + 1; i <= num_object + num_object_sent; ++i)
201 object_begin[i] += object_offset;
202
203 if (receive_num_child) {
204 /* receive num child */
205 MPI_Recv(&num_childs[num_object], num_object_sent, MPI_UNSIGNED, node, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
206
207 /* partial sum */
208 num_childs[num_object] += child_begin[num_object];
209 __gnu_parallel::partial_sum(num_childs.begin() + num_object, num_childs.begin() + num_object + num_object_sent, child_begin.begin() + num_object + 1);
210 num_childs[num_object] -= child_begin[num_object];
211 }
212
213 num_object += num_object_sent;
214 }

406 void simulate(mpi_it_t &iteration, quids::rule_t const *rule, mpi_it_t &next_iteration, mpi_sy_it_t &symbolic_iteration, MPI_Comm communicator, size_t max_num_object=0, quids::debug_t mid_step_function=[](const char*){}) {
407 /* get local size */
408 MPI_Comm localComm;
409 int rank, size, local_size;
410 MPI_Comm_size(communicator, &size);
411 MPI_Comm_rank(communicator, &rank);
412 MPI_Comm_split_type(communicator, MPI_COMM_TYPE_SHARED, rank, MPI_INFO_NULL, &localComm);
413 MPI_Comm_size(localComm, &local_size);
414
415
416
417
418
419
420 if (size == 1)
421 return quids::simulate(iteration, rule, next_iteration, symbolic_iteration, max_num_object, mid_step_function);
422
423
424 /* equalize objects */
425 if (!equalize_children) {
426 mid_step_function("equalize_object");
427 for (int max_equalize = std::log2(size); max_equalize >= 0; --max_equalize) {
428 /* check for condition */
429 size_t max_n_object = iteration.get_max_num_object_per_task(communicator);
430 float avg_n_object = iteration.get_avg_num_object_per_task(communicator);
431 float inbalance = ((float)max_n_object - avg_n_object)/max_n_object;
432
433 // debug:
434 if (rank == 0)
435 std::cerr << "\tmax=" << max_n_object << ", avg=" << avg_n_object << ", inbalance=" << inbalance << "\n";
436
437 if (max_n_object < min_equalize_size || inbalance < equalize_inbalance)
438 break;
439
440 // debug:
441 if (rank == 0)
442 std::cerr << "\tequalize\n";
443
444 /* actually equalize */
445 iteration.equalize(communicator);
446 }
447 }
448
449
450 /* start actual simulation */
451 iteration.compute_num_child(rule, mid_step_function);
452 iteration.truncated_num_object = iteration.num_object;
453
454
455 /* equalize symbolic objects */
456 if (equalize_children) {
457 mid_step_function("equalize_child");
458 for (int max_equalize = std::log2(size); max_equalize >= 0; --max_equalize) {
459 /* check for condition */
460 size_t max_n_object = iteration.get_max_num_object_per_task(communicator);
461 size_t max_n_child = iteration.get_max_num_symbolic_object_per_task(communicator);
462 float avg_n_child = iteration.get_avg_num_symbolic_object_per_task(communicator);
463 float inbalance = ((float)max_n_child - avg_n_child)/max_n_child;
464
465 // debug:
466 if (rank == 0)
467 std::cerr << "\tmax=" << max_n_child << ", avg=" << avg_n_child << ", inbalance=" << inbalance << "\n";
468
469 if (max_n_object < min_equalize_size || inbalance < equalize_inbalance)
470 break;
471
472 // debug:
473 if (rank == 0)
474 std::cerr << "\tequalize\n";
475
476 /* actually equalize */
477 iteration.equalize_symbolic(communicator);
478 iteration.truncated_num_object = iteration.num_object;
479 }
480 }
481
482
483 /* prepare truncate */
484 mid_step_function("truncate_symbolic - prepare");
485 iteration.prepare_truncate(mid_step_function);
486
487
488 /* max_num_object */
489 mid_step_function("truncate_symbolic");
490 if (max_num_object == 0) {
491 /* available memory */
492 size_t avail_memory = next_iteration.get_mem_size(localComm) + symbolic_iteration.get_mem_size(localComm) + quids::utils::get_free_mem();
493 size_t target_memory = avail_memory/local_size*(1 - quids::safety_margin);
494
495 /* actually truncate by binary search */
496 if (iteration.get_truncated_mem_size() > target_memory) {
497 size_t begin = 0, end = iteration.num_object;
498 while (end > begin + 1) {
499 size_t middle = (end + begin) / 2;
500 iteration.truncate(begin, middle, mid_step_function);
501
502 size_t used_memory = iteration.get_truncated_mem_size(begin);
503 if (used_memory < target_memory) {
504 target_memory -= used_memory;
505 begin = middle;
506 } else
507 end = middle;
508 }
509 }
510 } else
511 iteration.truncate(0, max_num_object/local_size, mid_step_function);
512
513
514 /* downsize if needed */
515 if (iteration.num_object > 0) {
516 if (iteration.truncated_num_object < next_iteration.num_object)
517 next_iteration.resize(iteration.truncated_num_object);
518 size_t next_object_size = iteration.truncated_num_object*iteration.get_object_length()/iteration.num_object;
519 if (next_object_size < next_iteration.objects.size())
520 next_iteration.allocate(next_object_size);
521 }
522
523
524 /* rest of the simulation */
525 iteration.generate_symbolic_iteration(rule, symbolic_iteration, mid_step_function);
526 symbolic_iteration.compute_collisions(communicator, mid_step_function);
527 symbolic_iteration.next_iteration_num_object = symbolic_iteration.num_object_after_interferences;
528
529
530 /* prepare truncate */
531 mid_step_function("truncate - prepare");
532 symbolic_iteration.prepare_truncate(mid_step_function);
533
534
535 /* second max_num_object */
536 mid_step_function("truncate");
537 if (max_num_object == 0) {
538 /* available memory */
539 size_t avail_memory = next_iteration.get_mem_size(localComm) + quids::utils::get_free_mem();
540 size_t target_memory = avail_memory/local_size*(1 - quids::safety_margin);
541
542 /* actually truncate by binary search */
543 if (symbolic_iteration.get_truncated_mem_size() > target_memory) {
544 size_t begin = 0, end = symbolic_iteration.num_object_after_interferences;
545 while (end > begin + 1) {
546 size_t middle = (end + begin) / 2;
547 symbolic_iteration.truncate(begin, middle, mid_step_function);
548
549 size_t used_memory = symbolic_iteration.get_truncated_mem_size(begin);
550 if (used_memory < target_memory) {
551 target_memory -= used_memory;
552 begin = middle;
553 } else
554 end = middle;
555 }
556 }
557 } else
558 symbolic_iteration.truncate(0, max_num_object/local_size, mid_step_function);
559
560
561 /* finalize simulation */
562 symbolic_iteration.finalize(rule, iteration, next_iteration, mid_step_function);
563 next_iteration.normalize(communicator, mid_step_function);
564
565 MPI_Comm_free(&localComm);
566 }

569 get the truncated memory size
570 */
571 size_t mpi_iteration::get_truncated_mem_size(size_t begin_num_object) const {
572 static const size_t iteration_memory_size = ITERATION_MEMORY_SIZE;
573
574 static const float hash_map_size = HASH_MAP_OVERHEAD*2*sizeof(size_t);
575 static const size_t symbolic_iteration_memory_size = SYMBOLIC_ITERATION_MEMORY_SIZE + MPI_SPECIFIC_SYMBOLIC_ITERATION_MEMORY_SIZE;
576 static const size_t mpi_symbolic_iteration_memory_size = MPI_SYMBOLIC_ITERATION_MEMORY_SIZE;
577
578 size_t mem_size = iteration_memory_size*(truncated_num_object - begin_num_object);
579 for (size_t i = begin_num_object; i < truncated_num_object; ++i) {
580 size_t oid = truncated_oid[i];
581
582 mem_size += object_begin[oid + 1] - object_begin[oid];
583 mem_size += num_childs[oid]*(symbolic_iteration_memory_size + mpi_symbolic_iteration_memory_size + hash_map_size);
584 }
585
586 return mem_size;
587 }

592 void mpi_symbolic_iteration::compute_collisions(MPI_Comm communicator, quids::debug_t mid_step_function) {
593 int size, rank;
594 MPI_Comm_size(communicator, &size);
595 MPI_Comm_rank(communicator, &rank);
596
597 if (size == 1)
598 return quids::symbolic_iteration::compute_collisions(mid_step_function);
599
600 int num_threads;
601 #pragma omp parallel
602 #pragma omp single
603 num_threads = omp_get_num_threads();
604
605 int const n_segment = size*num_threads;
606 int const num_bucket = quids::utils::nearest_power_of_two(load_balancing_bucket_per_thread*n_segment);
607 size_t const offset = 8*sizeof(size_t) - quids::utils::log_2_upper_bound(num_bucket);
608
609 std::vector<int> load_balancing_begin(n_segment + 1, 0);
610 std::vector<size_t> partition_begin(num_bucket + 1, 0);
611 std::vector<size_t> total_partition_begin(num_bucket + 1, 0);
612
613 std::vector<int> local_disp(n_segment + 1);
614 std::vector<int> local_count(n_segment);
615 std::vector<int> global_disp(n_segment + 1, 0);
616 std::vector<int> global_count(n_segment);
617
618 std::vector<int> send_disp(size + 1);
619 std::vector<int> send_count(size);
620 std::vector<int> receive_disp(size + 1);
621 std::vector<int> receive_count(size);
622
623 mid_step_function("compute_collisions - prepare");
624 mpi_resize(num_object);
625
626
627
628
629
630 /* !!!!!!!!!!!!!!!!
631 partition
632 !!!!!!!!!!!!!!!! */
633 quids::utils::parallel_generalized_partition_from_iota(&next_oid[0], &next_oid[0] + num_object, 0,
634 partition_begin.begin(), partition_begin.end(),
635 [&](size_t const oid) {
636 return hash[oid] >> offset;
637 });
638
639 /* generate partitioned hash */
640 #pragma omp parallel for
641 for (size_t id = 0; id < num_object; ++id) {
642 size_t oid = next_oid[id];
643
644 partitioned_mag[id] = magnitude[oid];
645 partitioned_hash[id] = hash[oid];
646 }

654 /* !!!!!!!!!!!!!!!!

851 void mpi_iteration::normalize(MPI_Comm communicator, quids::debug_t mid_step_function) {

853 normalize

856
857 node_total_proba = 0;

859

861 for (size_t oid = 0; oid < num_object; ++oid)

881 /*
882 equalize the number of objects across nodes
883 */
884 void mpi_iteration::equalize(MPI_Comm communicator) {
885 MPI_Request request = MPI_REQUEST_NULL;
886
887 MPI_Comm localComm;
888 int rank, size, local_size;
889 MPI_Comm_size(communicator, &size);
890 MPI_Comm_rank(communicator, &rank);
891 MPI_Comm_split_type(communicator, MPI_COMM_TYPE_SHARED, rank, MPI_INFO_NULL, &localComm);
892 MPI_Comm_size(localComm, &local_size);
893
894 int this_pair_id;
895 if (rank == 0) {
896 /* gather sizes */
897 std::vector<size_t> sizes(size, 0);
898 MPI_Gather(&num_object, 1, MPI_UNSIGNED_LONG_LONG, &sizes[0], 1, MPI_UNSIGNED_LONG_LONG, 0, communicator);
899
900 /* compute pair_id*/
901 std::vector<int> pair_id(size, 0);
902 utils::make_equal_pairs(&sizes[0], &sizes[0] + size, &pair_id[0]);
903
904 /* scatter pair_id */
905 MPI_Scatter(&pair_id[0], 1, MPI_INT, &this_pair_id, 1, MPI_INT, 0, communicator);
906 } else {
907 /* gather sizes */
908 MPI_Gather(&num_object, 1, MPI_UNSIGNED_LONG_LONG, NULL, 1, MPI_UNSIGNED_LONG_LONG, 0, communicator);
909
910 /* scatter pair_id */
911 MPI_Scatter(NULL, 1, MPI_INT, &this_pair_id, 1, MPI_INT, 0, communicator);
912 }
913
914 /* get available memory */
915 MPI_Barrier(localComm);
916 size_t total_iteration_size, iteration_size = quids::iteration::get_mem_size();
917 MPI_Allreduce(&iteration_size, &total_iteration_size, 1, Proba_MPI_Datatype, MPI_SUM, localComm);
918 size_t avail_memory = ((quids::utils::get_free_mem() + total_iteration_size)/local_size - iteration_size)*(1 - quids::safety_margin);
919 MPI_Barrier(localComm);
920
921 /* skip if this node is alone */
922 if (this_pair_id == rank)
923 return;
924
925 /* get the number of objects of the respective pairs */
926 size_t other_num_object;
927 MPI_Isend(&num_object, 1, MPI_UNSIGNED_LONG_LONG, this_pair_id, 0 /* tag */, communicator, &request);
928 MPI_Isend(&num_object, 1, MPI_UNSIGNED_LONG_LONG, this_pair_id, 0 /* tag */, communicator, &request);
929
930 MPI_Recv(&other_num_object, 1, MPI_UNSIGNED_LONG_LONG, this_pair_id, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
931 MPI_Recv(&other_num_object, 1, MPI_UNSIGNED_LONG_LONG, this_pair_id, 0 /* tag */, communicator, MPI_STATUS_IGNORE);
932
933 /* equalize amoung pairs */
934 if (num_object > other_num_object) {
935 size_t num_object_sent = (num_object - other_num_object) / 2;
936 send_objects(num_object_sent, this_pair_id, communicator, false);
937 } else if (num_object < other_num_object)
938 receive_objects(this_pair_id, communicator, false, avail_memory);
939 }

972

1009 } else if (num_symbolic_object < other_num_object)
1010 receive_objects(this_pair_id, communicator, true, avail_memory);
